// Load AI models
async function loadModels() {
try {
statusBox.textContent = 'Loading AI models...';
statusBox.className = 'status-box status-loading';

await faceapi.nets.tinyFaceDetector.loadFromUri('https://cdn.jsdelivr.net/npm/@vladmandic/face-api@1.7.12/model');

}
}

// Start camera
async function startCamera() {
try {
statusBox.textContent = 'Starting camera...';
statusBox.className = 'status-box status-scanning';

videoStream = await navigator.mediaDevices.getUserMedia({
video: {
facingMode: 'user',
width: { ideal: 1280 },
height: { ideal: 720 }
}
});

video.srcObject = videoStream;
video.style.display = 'block';
video.classList.add('active');

btnCamera.style.display = 'none';
btnCapture.style.display = 'block';

statusBox.textContent = 'Position your face clearly, then click Capture Face';
statusBox.className = 'status-box status-ready';

// Start face detection preview
startFaceDetectionPreview();
} catch (err) {
console.error('Camera error:', err);
statusBox.textContent = 'Camera access denied. Please allow permissions.';
statusBox.className = 'status-box status-error';
}
}

// Start face detection preview
async function startFaceDetectionPreview() {
if (!videoStream) return;

try {
const detection = await faceapi.detectSingleFace(
video,
new faceapi.TinyFaceDetectorOptions({ inputSize: 640, scoreThreshold: 0.3 })
);

if (detection) {
statusBox.textContent = 'Face detected! Click Capture Face';
statusBox.className = 'status-box status-ready';
} else {
statusBox.textContent = 'Position your face in the camera';
statusBox.className = 'status-box status-scanning';
}
} catch (err) {
console.log('Detection preview error:', err);
}

if (videoStream) {
setTimeout(startFaceDetectionPreview, 1000);
}
}

// Capture face
async function captureFace() {
try {
statusBox.textContent = 'Verifying face...';
statusBox.className = 'status-box status-scanning';
btnCapture.disabled = true;

// Get face descriptor
const detection = await faceapi.detectSingleFace(
video,
new faceapi.TinyFaceDetectorOptions({ inputSize: 640, scoreThreshold: 0.3 })
).withFaceLandmarks()
.withFaceDescriptor();

if (!detection) {
statusBox.textContent = 'No face detected! Try better lighting';
statusBox.className = 'status-box status-error';
btnCapture.disabled = false;
return;
}

// Get user's stored face descriptor
const userFace = (await db.ref('users/' + currentUser.uid + '/faceDescriptor').once('value')).val();
if (!userFace) {
statusBox.textContent = 'Face data not found!';
statusBox.className = 'status-box status-error';
btnCapture.disabled = false;
return;
}

// Compare faces
const faceMatcher = new faceapi.FaceMatcher([{
descriptor: userFace,
label: currentUser.uid
}], 0.75);

const result = faceMatcher.findBestMatch(detection.descriptor);

if (result.label !== currentUser.uid) {
statusBox.textContent = '❌ Face not recognized! Please try again.';
statusBox.className = 'status-box status-error';
btnCapture.disabled = false;
return;
}

// Face verified - show NFC instructions
faceScan.style.display = 'none';
nfcInstructions.style.display = 'block';
statusBox.textContent = 'Face verified! Tap NFC tag or scan QR code';
statusBox.className = 'status-box status-ready';

} catch (err) {
console.error('Face verification error:', err);
statusBox.textContent = 'Error: ' + err.message;
statusBox.className = 'status-box status-error';
btnCapture.disabled = false;
}
}

// Save sign in to database
function saveSignIn(action) {
const timestamp = firebase.database.ServerValue.TIMESTAMP;
const ref = db.ref('signins').push();

ref.set({
userId: currentUser.uid,
action: action,
siteId: currentSite,
timestamp: timestamp,
location: 'Wellwood Gardens, Redhills, Exeter, EX4 1HY',
method: 'FACE'
});
}
