// Initialize Firebase
firebase.initializeApp(firebaseConfig);
const auth = firebase.auth();
const db = firebase.database();

// DOM Elements
const authForm = document.getElementById('authForm');
const faceEnrollment = document.getElementById('faceEnrollment');
const statusMsg = document.getElementById('statusMsg');
const captureBtn = document.getElementById('captureBtn');
const enrollmentStatus = document.getElementById('enrollmentStatus');

let videoStream = null;
let modelsLoaded = false;

// Show status message
function showStatus(msg, type) {
statusMsg.textContent = msg;
statusMsg.className = 'status-msg ' + type;
statusMsg.style.display = 'block';
setTimeout(() => { statusMsg.style.display = 'none'; }, 4000);
}

// Toggle between login and signup
function toggleForm() {
if (document.getElementById('authForm').style.display === 'none') {
document.getElementById('authForm').style.display = 'block';
document.getElementById('faceEnrollment').style.display = 'none';
} else {
document.getElementById('authForm').style.display = 'none';
document.getElementById('faceEnrollment').style.display = 'block';
}
}

// Handle authentication
async function handleAuth() {
const email = document.getElementById('email').value;
const password = document.getElementById('password').value;

const isLogin = document.getElementById('authForm').style.display === 'block';

try {
if (isLogin) {
await auth.signInWithEmailAndPassword(email, password);
showStatus('✅ Successfully logged in!', 'success');
// Check if face is enrolled
const user = auth.currentUser;
const enrolled = (await db.ref('users/' + user.uid + '/faceEnrolled').once('value')).val();
if (enrolled) {
window.location.href = 'scan.html';
} else {
document.getElementById('authForm').style.display = 'none';
document.getElementById('faceEnrollment').style.display = 'block';
}
} else {
await auth.createUserWithEmailAndPassword(email, password);
showStatus('✅ Account created! Please enroll your face.', 'success');
document.getElementById('authForm').style.display = 'none';
document.getElementById('faceEnrollment').style.display = 'block';
}
} catch (err) {
showStatus('❌ Error: ' + err.message, 'error');
}
}

// Start camera
async function startCamera() {
try {
videoStream = await navigator.mediaDevices.getUserMedia({
video: {
facingMode: 'user',
width: { ideal: 1280 },
height: { ideal: 720 }
}
});

document.getElementById('video').srcObject = videoStream;
captureBtn.disabled = false;
showStatus('✅ Camera started! Position your face clearly', 'success');
} catch (err) {
showStatus('❌ Camera error: ' + err.message, 'error');
}
}

// Capture face
async function captureFace() {
try {
showStatus('Capturing face...', 'success');
enrollmentStatus.textContent = 'Capturing face...';

// Load face recognition models
if (!modelsLoaded) {
await faceapi.nets.tinyFaceDetector.loadFromUri('https://cdn.jsdelivr.net/npm/@vladmandic/face-api@1.7.12/model');
await faceapi.nets.faceLandmark68Net.loadFromUri('https://cdn.jsdelivr.net/npm/@vladmandic/face-api@1.7.12/model');
await faceapi.nets.faceRecognitionNet.loadFromUri('https://cdn.jsdelivr.net/npm/@vladmandic/face-api@1.7.12/model');
modelsLoaded = true;
}

// Detect face
const detection = await faceapi.detectSingleFace(
document.getElementById('video'),
new faceapi.TinyFaceDetectorOptions({ inputSize: 640, scoreThreshold: 0.3 })
).withFaceLandmarks()
.withFaceDescriptor();

if (!detection) {
showStatus('❌ No face detected! Try better lighting', 'error');
enrollmentStatus.textContent = 'No face detected';
return;
}

// Save face descriptor
const user = auth.currentUser;
await db.ref('users/' + user.uid).update({
faceDescriptor: Array.from(detection.descriptor),
faceEnrolled: true,
createdAt: firebase.database.ServerValue.TIMESTAMP
});

showStatus('✅ Face enrolled successfully!', 'success');
enrollmentStatus.textContent = 'Face enrolled successfully!';

// Redirect to scan page
setTimeout(() => {
window.location.href = 'scan.html';
}, 2000);
} catch (err) {
console.error('Face capture error:', err);
showStatus('❌ Error: ' + err.message, 'error');
enrollmentStatus.textContent = 'Error: ' + err.message;
}
}

// Initialize
if (auth.currentUser) {
// Check if face is enrolled
const user = auth.currentUser;
db.ref('users/' + user.uid + '/faceEnrolled').once('value', (snap) => {
if (snap.val()) {
window.location.href = 'scan.html';
} else {
document.getElementById('authForm').style.display = 'none';
document.getElementById('faceEnrollment').style.display = 'block';
}
});
}
